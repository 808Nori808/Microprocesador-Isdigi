.data 
valor: .word 12

.text
la x9,valor #Cargamos la dirección de memoria inicial para ir guardando los valores de la serie de Fibonacci
lw x1,valor #Cargamos el valor del número de iteraciones almacenado en memoria

# Reinicio de los registros y calculo del primer valor
addi x3,x0,0 #Registro para almacenamiento de 1 de los valores a sumar en Fibonacci	
addi x4,x0,0 #Registro para almacenar el otro valor
addi x6,x0,0 #Registro para almacenar el resultado de la suma
addi x5,x0,0 #Registro de apoyo para almacenar valores
addi x7,x0,0 #Registro para saber el número de iteraciones que lleva el programa hechas en cada momento
addi x8,x0,1 #Registro para habilitar el cálculo del segundo valor
beq x0,x0,sumar

segundovalor:
	addi x4,x0,1
    add x6,x5,x6
    beq x0,x0,sumar


resetea:
	beq x7,x8,segundovalor
	sub x3,x3,x3
    sub x4,x4,x4
	add x3,x0,x6
    add x4,x0,x5
    beq x0,x0,desplazar
desplazar: 
    sub x6,x6,x6
	add x6,x5,x6
    beq x0,x0,sumar    
sumar:
	add x5,x3,x4
    addi x7,x7,1
    addi x9,x9,4
    sw x5,0(x9)
    bne x7,x1,resetea
